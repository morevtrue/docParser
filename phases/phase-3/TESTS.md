## Как протестировать через Python

```python
import requests, zipfile, io, json

files = [('files', (f.split('/')[-1], open(f, 'rb'), 'application/vnd.openxmlformats-officedocument.wordprocessingml.document'))
         for f in [
             'Основной набор/Документы от заказчика/01_Запрос_ТКП_от_заказчика.docx',
             'Основной набор/Документы поставщика/01_Карточка_поставщика.docx',
             'Основной набор/Документы поставщика/02_Коммерческие_условия_и_цены_поставщика.docx',
             'Шаблоны/01_Анкета_участника_шаблон.docx',
             'Шаблоны/02_Заявка_на_участие_в_закупке_шаблон.docx',
             'Шаблоны/03_Предложение_о_цене_договора_шаблон.docx',
         ]]
r = requests.post('http://localhost:8000/api/process', files=files)
zf = zipfile.ZipFile(io.BytesIO(r.content))
print(zf.namelist())
print(json.loads(zf.read('report.json')))
```
